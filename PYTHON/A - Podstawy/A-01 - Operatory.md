% [modulo]- zwraca ==resztę== z dzielenia dwóch liczb
	a = 5
	b = 2
	a % b = 5 / 2 = 2x2+==1== = ==1==
// - dzielenie całkowite
	a = 7
	b = 2
	a//b= 7 // 2 = ==**3**== + 0.5 = ==3==
\** - potęgowanie
	a = 2
	b = 3
	a \** b = 2^3 = 8 
	pow(a, b) = 2^3 = 8
Zmienianie wartości zmiennej
	a = +1
	a = -3
	a = \*2
	a = /4
Operatory porównania
	a == b
	a >= b
	a < b
	a != b (*różne*)
^ - XOR bitowy (alternatywa wkluczająca na poziomie bitów)
	a = 5 = 101
	b = 3 = 011
	1 - jeśli bity są różne
	0 - jeśli bity są identyczne
	a ^ b = 101 ^ 011 = 110 = 6
